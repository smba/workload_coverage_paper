% Macro for research questions
\newcommand{\RQ}[2]{
	\hypertarget{rq:#1}{
		\vspace{0.5em}
		\bgroup
		\def\arraystretch{1.3}%  1 is the default, change whatever you need
		\begin{tabularx}{\linewidth}{p{0.12\linewidth}?p{0.75\linewidth}}
			\rowcolor{white!5!white}
			$\text{RQ}_\text{#1}$ & \textit{#2} \\
		\end{tabularx}
		\egroup
	}
}
